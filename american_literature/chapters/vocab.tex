\chapter{Vocabulary}
\section{Unit 9}
\begin{description}
  \item[abate]{to make less in amount}
  \item[adulation]{praise or flattery in excess}
  \item[anathema]{an object of intense dislike}
  \item[astute]{shrewd, crafty, having practical wisdom: "street smarts"}
  \item[avarice]{a greedy desire, usually having to do with money}
  \item[culpable]{worth of blame, condemnation}
  \item[dilatory]{tending to delay}
  \item[egregious]{conspicuous, standing out form the mass}
  \item[equivocate]{to speak or act in a way that allows for more than one
    interpretation, ambiguity}
  \item[evanescent]{vanishing, soon passing}
  \item[irresolute]{unable to make up ones own mind}
  \item[nebulous]{cloudlike, resembling a cloud}
  \item[novice]{one who is new to something}
  \item[penury]{extreme poverty}
  \item[pretentious]{done for show, striving to make a big impression}
  \item[recapitulate]{to review a series of facts, to sum up}
  \item[resuscitate]{to revive, to bring back}
  \item[slovenly]{untidy, dirty, careless}
  \item[supposition]{something that is assumed or taken without conclusive
    evidence}
  \item[torpid]{sluggish, slow}
\end{description}
